public class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int n = A.length;
        for (int i = 0; i < n; i++) {
            int left = 0, right = n - 1;
            while (left < right) {
                //å·¦å³æŒ‡é’ˆéƒ½æ˜¯0æˆ–éƒ½æ˜¯1
                if (A[i][left] == A[i][right]) {
                    A[i][left] ^= 1;
                    A[i][right] ^= 1;
                }
                left++;
                right--;
            }
            //å·¦å³æŒ‡é’ˆæŒ‡å‘åŒä¸€ä¸ªæ•°
            if (left == right) {
                A[i][left] ^= 1;
            }
        }
        return A;
    }
