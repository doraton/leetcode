class Solution(object):
    def generateMatrix(self, n):
        #å¥å£®æ€§åˆ†æ
        if n == 0: return []
        res = [[0] * n for i in range(n)]
        #åˆå§‹åŒ–å·¦ã€å³ã€ä¸Šã€ä¸‹çš„è¾¹ç•Œå€¼
        left, right, up, down = 0, n - 1, 0, n - 1
        #è¡¨ç¤ºå½“å‰éå†åˆ°çš„ä½ç½®
        x, y = 0, 0
        #è¡¨ç¤ºç§»åŠ¨æ–¹å‘å·¦
        dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        #å½“å‰ç§»åŠ¨æ–¹å‘çš„ä¸‹æ ‡
        cur_d = 0
        count = 0
        while count != n * n:
            res[x][y] = count + 1
            count += 1
            if cur_d == 0 and y == right:
                cur_d += 1
                up += 1
            elif cur_d == 1 and x == down:
                cur_d += 1
                right -= 1
            elif cur_d == 2 and y == left:
                cur_d += 1
                down -= 1
            elif cur_d == 3 and x == up:
                cur_d += 1
                left += 1
            cur_d %= 4
            x += dirs[cur_d][0]
            y += dirs[cur_d][1]
        return res
